### Algorithm

https://leetcode.com/problems/best-time-to-buy-and-sell-stock/

股票的最佳买入时期

```
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        
        if(prices.size() <= 0) return 0;
        
        int min = prices.at(0);
        int max = 0;
        for(int i = 0;i<prices.size();i++)
        {
              if(prices.at(i)<min)
                  min = prices.at(i);
              if( (prices.at(i) - min) > max)
                  max = prices.at(i) -min;
              
        }
        
        return max;
        
    }
};

```
